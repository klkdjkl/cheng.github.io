程式

我得想法:

  設定土壤檢測器再腳位A0，繼電器再腳位7。當Arduino讀取土壤檢測器值大於500時，繼電器高電壓，迴路通電(水泵啟動)，當讀取土壤檢測器值小於500時，繼電器低電壓，迴路中斷(水泵關閉)。

 

int sensorValue;

void setup() {

  Serial.begin(9600);

  pinMode(A0,INPUT);

  pinMode(7,OUTPUT);

}

 

void loop() {

  sensorValue=analogRead(A0);

    Serial.println(sensorValue);

  if (sensorValue > 500){

    digitalWrite(7, HIGH);

  }

  else {

    digitalWrite(7, LOW);

  }

  

 

}
